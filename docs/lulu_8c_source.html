113  else if (newCount == COUNT_DECREMENT)
114  newCount = (count > 0)? count - 1: 0;
115 
116  if (newCount == count)
117  return FALSE; // There is no change in count so skip the operation
118 
119  for (uint8_t i = 0; i < multiset->size; i++)
120  // if we are requested to delete an object from the multiset

125  multiset->items[i].id = NO_OBJECT;

133  multiset->items[i].id = obj;

142  }
